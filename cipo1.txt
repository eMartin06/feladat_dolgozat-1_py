cipo1=int(input('Adj meg az egyik cipő méretet! '))
cipo2=int(input('Adjon meg egy másik cipő méretet! '))

if cipo2 > cipo1:
    print(f'A cipő méret a másik cipő esetében nagyobb, {cipo2}')

if cipo2 > cipo1:
    print(f'A cipő méret az egyik cipő esetében kisebb, {cipo1}')
    
else:
    print('A két cipő egyenlő.')
